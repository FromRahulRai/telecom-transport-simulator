! Phase 1 - VLAN & Inter-VLAN Routing
! Device: SW-T1

hostname SW-T1

! VLAN Creation
vlan 10
 name VLAN_4G

vlan 20
 name VLAN_5G

! Access Port for 4G PC
interface fa0/1
 switchport mode access
 switchport access vlan 10
 no shutdown

! Access Port for 5G PC
interface fa0/2
 switchport mode access
 switchport access vlan 20
 no shutdown

! Trunk Port to Router
interface fa0/3
 switchport mode trunk
 no shutdown
